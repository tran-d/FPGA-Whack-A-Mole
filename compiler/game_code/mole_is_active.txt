mole_is_active: nop
	
	# a0: mole index
	# v0: 1 - active, 0 - not active

	addi $s0, $r0, 5    # find mole state in memory [(index * 5 )+ 1]
	mul $s0, $s0, $a0
	nop
	nop
	addi $s0, $s0, 1 

	lw $s1, 0($s0)      # load state
	beq $s1, $r0, 1
	j mole_is_active_true

	mole_is_active_false: nop
		addi $v0, $r0, 0
		jr $ra

	mole_is_active_true: nop
		addi $v0, $r0, 1
		jr $ra
