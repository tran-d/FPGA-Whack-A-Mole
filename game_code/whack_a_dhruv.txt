# led logic

led: nop

  bne $r10, $r5, 2    # hit bad dhruv
  addi $r28, $r28, 1  # UPDATE SCORE HERE (ADD)
  j jreturn

  bne $r10, $r5, 2    # hit good dhruv (sad)
  addi $r28, $r28, -1  # UPDATE SCORE HERE (SUBTRACT)
  j jreturn

  bne $r10, $r0, 2    # hit nothing (sad)
